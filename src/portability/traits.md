# The Trait System

```rust
/// Single digital push-pull output pin
pub trait OutputPin {
    /// Drives the pin low
    fn set_low(&mut self);

    /// Drives the pin high
    fn set_high(&mut self);
}
```

[rust-embedded/embedded-hal](https://github.com/rust-embedded/embedded-hal/blob/master/src/digital.rs)

impl<MODE> OutputPin for OutputGpio<MODE> {

        self.set_pin_low()

    }
}
